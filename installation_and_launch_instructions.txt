##INSTALLATION##

#If you do not have 'gem' and 'port'. Follow the following links to install
https://rubygems.org/pages/download
https://guide.macports.org/chunked/installing.macports.html

#Install system dependecies
gem install libv8 -v 3.11.8.17 -- --with-system-v8
sudo port install cairo
sudo port install xorg-libXt
sudo port install openssl
sudo port install libxml2

#Launch R console
sudo R

#Install CRAN dependencies
install.packages(c("shiny", "DT", "shinyjs", "shinyBS", "shinydashboard", "shinyFiles", "ggplot2", "infotheo", "WriteXLS", "rmarkdown", "VennDiagram", "UpSetR", "futile.logger"), repo="http://cran.rstudio.org", type="source", dependencies=T)

#Install Bioconductor dependencies
source("https://bioconductor.org/biocLite.R")
biocLite(c("limma", "sva", "swamp", "biomaRt"))

##LAUNCHING##

#Launch R console and go to PACKAGE diectory
R
setwd("<replace with installation directory>/eUTOPIA/")

#Call 'shiny' library
library(shiny)

#Launch shiny app
runApp("eUTOPIA-app/")
